<script>

  const moves = {

    r: { label: 'Rock',     emoji: '🪨' },

    p: { label: 'Paper',    emoji: '📄' },

    s: { label: 'Scissors', emoji: '✂️' }

  };

  const cpuKeys = ['r', 'p', 's'];

  let w = 0, t = 0, l = 0;


  function getResult(user, cpu) {

    if (user === cpu) return 'tie';

    if (

      (user === 'r' && cpu === 's') ||

      (user === 'p' && cpu === 'r') ||

      (user === 's' && cpu === 'p')

    ) return 'win';

    return 'lose';

  }


  function play(user) {

    const cpu    = cpuKeys[Math.floor(Math.random() * 3)];

    const result = getResult(user, cpu);


    let msg = '', color = '';

    if (result === 'win')       { w++; msg = 'You win!';     color = '#2a9d5c'; }

    else if (result === 'tie')  { t++; msg = "It's a tie!";  color = '#333';    }

    else                        { l++; msg = 'You lose!';    color = '#e24b4a'; }


    document.getElementById('wins').textContent   = w;

    document.getElementById('ties').textContent   = t;

    document.getElementById('losses').textContent = l;


    document.getElementById('arena').innerHTML = `

      <div class="vs-row">

        <div class="player">

          <div class="player-emoji">${moves[user].emoji}</div>

          <div class="player-label">You</div>

        </div>

        <div class="vs-text">vs</div>

        <div class="player">

          <div class="player-emoji">${moves[cpu].emoji}</div>

          <div class="player-label">Computer</div>

        </div>

      </div>

      <div class="result-text" style="color:${color};">${msg}</div>

      <div class="match-text">${moves[user].label} vs ${moves[cpu].label}</div>

    `;

  }


  function quit() {

    const total = w + t + l;

    let emoji, title, sub;


    if (total === 0) {

      emoji = '👋'; title = 'No rounds played'; sub = 'Come back anytime!';

    } else if (w > l) {

      emoji = '🏆'; title = 'You won overall!';      sub = `${total} round${total !== 1 ? 's' : ''} played`;

    } else if (l > w) {

      emoji = '😬'; title = 'Computer won overall';  sub = `${total} round${total !== 1 ? 's' : ''} played`;

    } else {

      emoji = '🤝'; title = 'Overall tie!';           sub = `${total} round${total !== 1 ? 's' : ''} played`;

    }


    document.getElementById('summary-emoji').textContent = emoji;

    document.getElementById('summary-title').textContent = title;

    document.getElementById('summary-sub').textContent   = sub;

    document.getElementById('s-wins').textContent        = w;

    document.getElementById('s-ties').textContent        = t;

    document.getElementById('s-losses').textContent      = l;


    document.getElementById('game').style.display    = 'none';

    document.getElementById('summary').style.display = 'block';

  }


  function restart() {

    w = 0; t = 0; l = 0;

    document.getElementById('wins').textContent    = 0;

    document.getElementById('ties').textContent    = 0;

    document.getElementById('losses').textContent  = 0;

    document.getElementById('arena').innerHTML     = '<div class="placeholder">Choose your move above</div>';

    document.getElementById('summary').style.display = 'none';

    document.getElementById('game').style.display    = 'block';

  }

</script>
